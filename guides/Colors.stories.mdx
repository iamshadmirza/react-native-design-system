# Colors
These are all the colors that you can use.

```js
//usage
import { colors } from 'react-native-design-system';

theme.colors.primary = colors['lime-A700'];
```
<ColorPalette>
  <FlatList
    data={Object.keys(colors).filter(item => typeof colors[item] === 'string')}
    keyExtractor={(hex, index) => index.toString()}
    renderItem={({ item: shade }) => {
      return (
        <ColorItem
          title={shade}
          subtitle={shade.toUpperCase().split('-').join(' ')}
          colors={[colors[shade]]}
        />
      )
    }}
  />
</ColorPalette>
