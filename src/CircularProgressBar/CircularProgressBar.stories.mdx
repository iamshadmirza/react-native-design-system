# CircularProgressBar

Animated circular progress bar for cross plateform.

# Usage

```js
import {CircularProgressBar} from 'react-native-design-system';

<CircularProgressBar
  activeColor="darkviolet"
  passiveColor="darkgray"
  baseColor="white"
  width={50}
  percent={35}
  radius={100}
  duration={1200}>
  <Text>Wow!</Text>
</CircularProgressBar>;
```
